const STORAGE_KEY = 'fsbConversationMessages';
const DEFAULT_CAP = 50;
const DEFAULT_DEBOUNCE_MS = 200;
const ENVELOPE_VERSION = 1;

/**
 * @returns {object} { v: 1, byConv: {}, lru: [] }
 */
function emptyEnvelope();

/**
 * @param {object} envelope
 * @returns {boolean} true if envelope shape is { v: 1, byConv: object, lru: array }
 */
function isValidEnvelope(envelope);

/**
 * Touch the LRU order for a convId -- moves it to head (MRU position).
 * @param {object} envelope -- mutated in place
 * @param {string} convId
 */
function _touchLru(envelope, convId);

/**
 * Drop entries beyond `cap` from tail. Idempotent. Mutates in place.
 * Also reaps byConv entries not present in the lru array (defense vs corruption).
 * @param {object} envelope
 * @param {number} cap
 */
function _enforceLruCap(envelope, cap);

/**
 * Append a message to the per-convId log inside the envelope.
 * Lazy-creates the per-convId log on first call. Updates lastWriteAt + LRU.
 * @param {object} envelope -- mutated in place
 * @param {string} convId
 * @param {{role: 'user'|'assistant', content: string, timestamp: number, kind: 'text'|'progress'|'tool'|'error'}} msg
 * @returns {boolean} true on success, false on invalid input
 */
function appendMessage(envelope, convId, msg);

/**
 * Peek-only fetch of messages for convId. Returns empty array on missing.
 * @param {object} envelope
 * @param {string} convId
 * @returns {Array} messages (may be empty)
 */
function getMessages(envelope, convId);

/**
 * Idempotent drop: removes byConv[convId] + lru position.
 * @param {object} envelope -- mutated in place
 * @param {string} convId
 */
function dropConversationMessages(envelope, convId);

/**
 * Factory producing a per-convId debouncer.
 *
 * Per the Phase 12 D-03 contract:
 *   - Each call to `schedule(convId, callback)` clears any pending timer for
 *     that convId AND replaces it with a fresh 200ms timer (clear-and-replace).
 *   - Timer fires the supplied callback exactly once. Callback may be async.
 *   - `flush(convId)` forces synchronous timer cancel + immediate callback fire.
 *   - `flushAll()` iterates over all pending convIds + flushes each in turn.
 *   - `cancel(convId)` clears the timer without firing the callback (used by
 *     dropConversationMessages to defeat resurrection after drop, per
 *     EC-05).
 *
 * Implementation note: dependency-injected `setTimeoutFn` + `clearTimeoutFn`
 * so smoke tests can synchronously advance simulated time. Defaults to
 * globalThis setTimeout / clearTimeout.
 *
 * @param {{debounceMs?: number, setTimeoutFn?: function, clearTimeoutFn?: function}} [opts]
 * @returns {{
 *   schedule: (convId: string, cb: () => void|Promise<void>) => void,
 *   flush:    (convId: string) => Promise<void>,
 *   flushAll: () => Promise<void>,
 *   cancel:   (convId: string) => void,
 *   _hasPending: (convId: string) => boolean
 * }}
 */
function createDebouncer(opts);

            ? opts.debounceMs
            : DEFAULT_DEBOUNCE_MS;
          var _setTimeoutFn = (typeof opts.setTimeoutFn === 'function')
            ? opts.setTimeoutFn
            : function (fn, ms) { return setTimeout(fn, ms); };
          var _clearTimeoutFn = (typeof opts.clearTimeoutFn === 'function')
            ? opts.clearTimeoutFn
            : function (id) { return clearTimeout(id); };
          var _pending = {};

          function schedule(convId, cb) {
            var key = _normalizeConvId(convId);
            if (key === null || typeof cb !== 'function') return;
            var existing = _pending[key];
            if (existing && existing.timerId !== null) {
              _clearTimeoutFn(existing.timerId);
            }
            var entry = { timerId: null, cb: cb };
            _pending[key] = entry;
            entry.timerId = _setTimeoutFn(function () {
              entry.timerId = null;
              delete _pending[key];
              try { Promise.resolve(cb()).catch(function () {}); }
              catch (_e) { /* swallow */ }

            if (key === null) return;

            if (entry.timerId !== null) _clearTimeoutFn(entry.timerId);
            delete _pending[key];
            try { await Promise.resolve(entry.cb()); }
            catch (_e) { /* swallow */ }

            var keys = Object.keys(_pending);
            for (var i = 0; i < keys.length; i++) {

            if (!entry) return;
            if (entry.timerId !== null) _clearTimeoutFn(entry.timerId);
            delete _pending[key];
          }

          function _hasPending(convId) {
            var key = _normalizeConvId(convId);
            if (key === null) return false;
            return _pending[key] !== undefined;
          }

          return {

            flush: flush,
            flushAll: flushAll,
            cancel: cancel,
