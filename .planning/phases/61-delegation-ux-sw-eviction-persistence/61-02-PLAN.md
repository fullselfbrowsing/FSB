};

  | 'completed' | 'stopped' | 'cancelled' | 'start_rejected'
  | 'wall_clock_timeout' | 'event_silence_timeout'
  | 'delegation_persistence_failed' | 'delegation_quota_exceeded' | 'delegation_ledger_corrupt'

  | 'hold_expired' | 'resume_ownership_lost' | 'daemon_restart_lost_run'
  | 'agent_protocol_drift' | 'tree_unsettled' | 'agent_failed' | 'unknown_failure';

type DelegationLedgerEntry = {
  v: 1;
  delegationId: string;
  sequence: number;
  timestamp: number;
  kind: 'init' | 'tool-call' | 'retry' | 'result' | 'state';
  state: DelegationState;
  title: string;
  detail: string | null;
  init: DelegationInit | null;
  tool: DelegationTool | null;
  retry: DelegationRetry | null;
  metrics: DelegationMetrics | null;
};

type DelegationSummary = DelegationMetrics & {
  state: 'completed' | 'failed' | 'stopped' | 'restart_lost';
};

type DelegationSnapshot = {
  v: 1;
  delegationId: string | null;
  provider: null | { id: 'claude-code'; label: string };
  state: DelegationState;
  connection: 'connected' | 'disconnected' | 'offline' | 'unpaired' | 'unsupported';
  entries: DelegationLedgerEntry[];
  summary: DelegationSummary | null;
  activeTab: null | { tabId: number; owned: boolean; canTakeControl: boolean };
  hold: null | { tabIds: number[]; expiresAt: number };
  terminal: null | { code: DelegationTerminalCode; releasedTabCount: number };
  hydrated: boolean;
};

type DelegationRuntimeEvent = {
  type: 'FSB_DELEGATION_UPDATED';
  snapshot: DelegationSnapshot;
  announceSequence: number | null;
};

For each entry kind, typed payloads are exact and nonmatching payloads are `null`: `init` uses only `init`; `tool-call` uses only `tool`; `retry` uses only `retry`; `result` may use only `metrics`; `state` has all four typed payloads `null`. `title`/`detail` are bounded presentation copy, never a machine-readable fallback. The implementation and UI must not parse them to recover client/model/session/tools, retry class, tool status, metrics, or terminal code. Init bounds are: profile/model/session 128 characters, client fixed to the canonical id/label, and at most 16 unique allowed-tool names of 96 characters. Tool call id/name are 128 characters, args summary 256, tab/duration are bounded nonnegative integers or `null`. Retry numbers are bounded nonnegative integers or `null`; an unrecognized provider class becomes literal `unknown`, never an unbounded raw string. Any daemon/provider diagnostic not in `DelegationTerminalCode` maps to `unknown_failure` while its bounded safe presentation remains in `detail`.

The controller owns one private `Map<delegationId, DelegationRecord>` plus at most one provisional start transaction before a server id exists. Transitions/watchdogs/promises are isolated per record; one run's observer, Stop, timeout, hold, or terminal failure cannot settle another. `hydrate()` restores all nonterminal records before subscriptions are enabled. `getSnapshot(id)` requires the exact server id (the background may separately return the closed idle/ready snapshot for `null`). Transitions follow the closed `DelegationState`; terminal snapshots use the exact `DelegationSummary`/closed terminal fields and never provider-native blobs. `announceSequence` is `null` for hydrate/non-entry state changes and the exact newly persisted sequence for a live entry. A repeated delivery of an already-rendered `(delegationId, announceSequence)` may be suppressed only by Plan 07 UI; persisted duplicate/conflicting sequences are corruption.
