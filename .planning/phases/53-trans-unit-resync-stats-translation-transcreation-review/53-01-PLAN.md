import re, pathlib
IDS = ["agents.meta.description","agents.schema.software.description","home.meta.description","support.faq.q.tools.a","support.schema.faq.tools.a"]
LOCALES = ["es","de","ja","zh-CN","zh-TW"]
def units(path):
  return len(re.findall(r'<trans-unit id="', pathlib.Path(path).read_text()))
def block(path, tid):
  t = pathlib.Path(path).read_text()
  m = re.search(rf'<trans-unit id="{re.escape(tid)}"[^>]*>(.*?)</trans-unit>', t, re.S)
  assert m, tid
  return m.group(1)
def source(body):
  return re.search(r'<source>(.*?)</source>', body, re.S).group(1)
def target(body):
  return re.search(r'<target[^>]*>(.*?)</target>', body, re.S).group(1)
def xids(s):
  return re.findall(r'<x id="([^"]+)"', s)
en = pathlib.Path('src/locale/messages.xlf')
for loc in LOCALES:
  p = f'src/locale/messages.{loc}.xlf'
  assert units(p) == 996, (loc, units(p))
  for tid in IDS:
    eb, lb = block(en, tid), block(p, tid)
    assert source(eb) == source(lb), (loc, tid, 'source drift still present')
    tgt = target(lb)
    assert tgt.strip(), (loc, tid)
    if tid == 'support.faq.q.tools.a':
      assert xids(source(eb)) == xids(tgt), (loc, tid, 'placeholder mismatch')
print('OK resync asserts')
